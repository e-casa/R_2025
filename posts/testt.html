Quali x Quanti: Teste T

Introdução:

O teste t é uma técnica estatística utilizada para comparar as médias de duas amostras ou avaliar se a média de uma única amostra é significativamente diferente de um valor de referência.

Exemplo básico com visualização:

amostra_A <- c(10, 12, 15, 11, 9)
amostra_B <- c(8, 9, 11, 13, 10)

# Criar data frame para visualização
dados_teste <- data.frame(
  valor = c(amostra_A, amostra_B),
  grupo = rep(c("Amostra A", "Amostra B"), each = 5)
)

# Boxplot com ggplot2
ggplot(dados_teste, aes(x = grupo, y = valor, fill = grupo)) +
  geom_boxplot(alpha = 0.7) +
  geom_jitter(width = 0.2, alpha = 0.5) +
  stat_summary(fun = mean, geom = "point", shape = 23, size = 3, fill = "red") +
  labs(title = "Comparação entre Amostras A e B",
       x = "Grupo", y = "Valores") +
  theme_minimal() +
  theme(legend.position = "none")
# Teste t
resultado <- t.test(amostra_A, amostra_B)
print(resultado)

if (resultado$p.value < alpha) {
  print("Rejeitar H0: As médias são diferentes.")
} else {

}

Exemplo IDH Regiões do Estado do Rio

cidades_rm <- c("Rio de Janeiro", "Niterói", "São Gonçalo", "Duque de Caxias", "Nova Iguaçu")
idh_rm <- c(0.85, 0.82, 0.78, 0.74, 0.76)

cidades_outras <- c("Petrópolis", "Campos dos Goytacazes", "Macaé", "Volta Redonda", "Angra dos Reis")
idh_outras <- c(0.80, 0.76, 0.72, 0.78, 0.75)

# Data frame completo
df_idh <- data.frame(
  Cidade = c(cidades_rm, cidades_outras),
  IDH = c(idh_rm, idh_outras),
  Regiao = rep(c("Região Metropolitana", "Outras Regiões"), each = 5)
)

# Gráfico de barras
ggplot(df_idh, aes(x = reorder(Cidade, IDH), y = IDH, fill = Regiao)) +
  geom_col() +
  coord_flip() +
  labs(title = "IDH por Cidade e Região",
       x = "Cidade", y = "IDH") +
  theme_minimal() +
  scale_fill_brewer(palette = "Set2")
# Boxplot comparativo
ggplot(df_idh, aes(x = Regiao, y = IDH, fill = Regiao)) +
  geom_boxplot(alpha = 0.7) +
  geom_jitter(width = 0.2, alpha = 0.6, size = 3) +
  stat_summary(fun = mean, geom = "point", shape = 23, size = 4, fill = "red") +
  labs(title = "Comparação do IDH entre Regiões",
       x = "Região", y = "IDH") +
  theme_minimal() +
  theme(legend.position = "none")
# Teste t
resultado <- t.test(idh_rm, idh_outras, paired = FALSE)
print(resultado)

Exemplo teste-t monitoria

notas_antes <- c(7.8, 8.2, 7.5, 7.1, 8.5, 6.9, 7.3, 8.0, 7.6, 8.1, 
                 6.8, 7.2, 7.7, 7.9, 7.4, 8.3, 7.2, 7.8, 7.6, 7.9, 
                 7.1, 8.0, 7.3, 7.7, 7.4, 8.1, 7.5, 7.8, 7.6, 8.0)
notas_depois <- c(8.5, 9.1, 8.2, 7.9, 9.3, 8.0, 8.4, 8.9, 8.2, 8.8, 
                  7.8, 8.3, 8.7, 8.9, 8.3, 9.2, 8.3, 8.7, 8.5, 8.9, 
                  8.1, 8.6, 8.2, 8.7, 8.4, 8.9, 8.5, 8.8, 8.6, 8.9)

dados <- data.frame(
  aluno = 1:30,
  Notas_Antes = notas_antes, 
  Notas_Depois = notas_depois
)

# Transformar para formato longo
dados_long <- dados %>%
  pivot_longer(cols = starts_with("Notas"), 
               names_to = "Periodo", 
               values_to = "Nota") %>%
  mutate(Periodo = str_replace(Periodo, "Notas_", ""))

# Boxplot comparativo
p1 <- ggplot(dados_long, aes(x = Periodo, y = Nota, fill = Periodo)) +
  geom_boxplot(alpha = 0.7) +
  geom_jitter(width = 0.2, alpha = 0.3) +
  labs(title = "Distribuição das Notas Antes e Depois da Monitoria",
       x = "Período", y = "Notas") +
  theme_minimal() +
  scale_fill_brewer(palette = "Pastel1") +
  theme(legend.position = "none")

# Gráfico de linhas individuais
p2 <- ggplot(dados, aes(x = 1, y = Notas_Antes)) +
  geom_segment(aes(x = 1, xend = 2, y = Notas_Antes, yend = Notas_Depois),
               alpha = 0.5, color = "gray50") +
  geom_point(aes(x = 1, y = Notas_Antes), color = "#F8766D", size = 2) +
  geom_point(aes(x = 2, y = Notas_Depois), color = "#00BFC4", size = 2) +
  scale_x_continuous(breaks = c(1, 2), labels = c("Antes", "Depois"), limits = c(0.5, 2.5)) +
  labs(title = "Evolução Individual das Notas",
       x = "Período", y = "Notas") +
  theme_minimal()

# Mostrar ambos os gráficos
grid.arrange(p1, p2, ncol = 2)
# Histograma da diferença
dados$diferenca <- dados$Notas_Depois - dados$Notas_Antes

ggplot(dados, aes(x = diferenca)) +
  geom_histogram(binwidth = 0.2, fill = "steelblue", color = "white", alpha = 0.7) +
  geom_vline(xintercept = mean(dados$diferenca), color = "red", linetype = "dashed", size = 1) +
  labs(title = "Distribuição da Diferença nas Notas (Depois - Antes)",
       x = "Diferença", y = "Frequência") +
  theme_minimal()
# Estatísticas descritivas
media_antes <- mean(dados$Notas_Antes)
media_depois <- mean(dados$Notas_Depois)
desvio_padrao_antes <- sd(dados$Notas_Antes)
desvio_padrao_depois <- sd(dados$Notas_Depois)

tabela_estatisticas <- data.frame(
  Estatística = c("Média", "Desvio-Padrão", "Mínimo", "Máximo"),
  Notas_Antes = c(media_antes, desvio_padrao_antes, min(dados$Notas_Antes), max(dados$Notas_Antes)),
  Notas_Depois = c(media_depois, desvio_padrao_depois, min(dados$Notas_Depois), max(dados$Notas_Depois))
)

print(tabela_estatisticas)
    Estatística Notas_Antes Notas_Depois

4        Máximo    8.500000    9.3000000
# Teste t pareado
resultado <- t.test(dados$Notas_Antes, dados$Notas_Depois, paired = TRUE)
print(resultado)
